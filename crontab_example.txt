# 每日数据更新定时任务示例
# 
# 使用方法：
# 1. 编辑 crontab: crontab -e
# 2. 添加以下行（根据你的项目路径调整）
# 3. 保存并退出

# 每天凌晨2点更新所有平台数据
0 2 * * * cd /Users/yansha/Documents/iLog && ./update_data.sh >> /Users/yansha/Documents/iLog/cron.log 2>&1

# 或者分别设置不同时间更新不同平台
# 每天凌晨1点更新简书数据
# 0 1 * * * cd /Users/yansha/Documents/iLog/jianshu-spider && ./run.sh >> /Users/yansha/Documents/iLog/cron.log 2>&1

# 每天凌晨1点30分更新B站数据
# 30 1 * * * cd /Users/yansha/Documents/iLog/bilibili-spider && ./run.sh >> /Users/yansha/Documents/iLog/cron.log 2>&1

# 每天凌晨2点更新豆瓣数据
# 0 2 * * * cd /Users/yansha/Documents/iLog/douban-spider && ./run.sh >> /Users/yansha/Documents/iLog/cron.log 2>&1
